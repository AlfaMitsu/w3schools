Use the margin property to center align the <h1> element.

    <!DOCTYPE html>
    <html>
    <head>
      <style>
        h1 {
          margin: auto;
        }
      </style>
    </head>
    <body>
      <h1>This is a heading</h1>
      <p>This is a paragraph</p>
      <p>This is a paragraph</p>
    </body>
    </html>
