Display the list items as inline elements.

    <!DOCTYPE html>
    <html>
    <head>
      <style>
        li {
          display: inline;
        }
      </style>
    </head>
    <body>
      <ul>
        <li>Apple</li>
        <li>Orange</li>
        <li>Pear</li>
      </ul>
    </body>
    </html>
