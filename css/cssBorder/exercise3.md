Use the correct border property to set the LEFT border to "dotted".

    <!DOCTYPE html>
    <html>
    <head>
      <style>
        p {
          border-left-style: dotted;
        }
      </style>
    </head>
    <body>
      <h1>This is a heading</h1>
      <p>This is a paragraph</p>
      <p>This is a paragraph</p>
    </body>
    </html>
