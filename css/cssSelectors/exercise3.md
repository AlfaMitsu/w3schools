Set the text color to red, for elements with class="colortext".

    <!DOCTYPE html>
    <html>
    <head>
      <style>
        .colortext {
          color: red;
        }
      </style>
    </head>
    <body>
      <h1>This is a heading</h1>
      <p>This is a paragraph</p>
      <p class="colortext">This is a paragraph</p>
      <p class="colortext">This is a paragraph</p>
    </body>
    </html>
